### Day 19 – Shell Scripting Mini Projects (Logs, Backups & Cron)

 

## Task 1 – Log Rotation Script

log_rotate.sh

#!/bin/bash

# Check argument

if [ $# -ne 1 ]; then

    echo "Usage: $0 <log_directory>"

    exit 1

fi

LOG_DIR=$1

# Check if directory exists

if [ ! -d "$LOG_DIR" ]; then

    echo "Error: Directory does not exist!"

    exit 1

fi

compressed_count=0

deleted_count=0

# Compress .log files older than 7 days

for file in $(find "$LOG_DIR" -type f -name "*.log" -mtime +7); do

    gzip "$file"

    if [ $? -eq 0 ]; then

        compressed_count=$((compressed_count+1))

    fi

done

# Delete .gz files older than 30 days

for file in $(find "$LOG_DIR" -type f -name "*.gz" -mtime +30); do

    rm -f "$file"

    if [ $? -eq 0 ]; then

        deleted_count=$((deleted_count+1))

    fi

done

echo "Compressed files: $compressed_count"

echo "Deleted files: $deleted_count"

## Task 2 – Server Backup Script

      backup.sh

           #!/bin/bash

if [ $# -ne 2 ]; then

    echo "Usage: $0 <source_directory> <backup_destination>"

    exit 1

fi

SOURCE=$1

DEST=$2

# Check if source exists

if [ ! -d "$SOURCE" ]; then

    echo "Error: Source directory does not exist!"

    exit 1

fi

# Create destination if not exists

mkdir -p "$DEST"

DATE=$(date +%Y-%m-%d)

ARCHIVE="$DEST/backup-$DATE.tar.gz"

# Create archive

tar -czf "$ARCHIVE" "$SOURCE"

 

# Verify creation

if [ $? -ne 0 ]; then

    echo "Backup failed!"

    exit 1

fi

# Get size

SIZE=$(du -h "$ARCHIVE" | awk '{print $1}')

echo "Backup created: $ARCHIVE"

echo "Size: $SIZE"

# Delete backups older than 14 days

find "$DEST" -type f -name "backup-*.tar.gz" -mtime +14 -exec rm -f {} \;

## Task 4 – Combined Maintenance Script

      maintenance.sh

           #!/bin/bash

LOGFILE="/var/log/maintenance.log"

log() {

    echo "$(date '+%Y-%m-%d %H:%M:%S') : $1" >> "$LOGFILE"

}

log "Maintenance started"

# Call log rotation

/log_rotate.sh /var/log/myapp >> "$LOGFILE" 2>&1

log "Log rotation completed"

# Call backup

/backup.sh /home/user/data /home/user/backups >> "$LOGFILE" 2>&1

log "Backup completed"

log "Maintenance finished"

 

      Cron Entry – Run Daily at 1 AM

           0 1 * * *  /maintenance.sh
