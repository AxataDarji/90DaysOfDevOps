## Task 3: Strict Mode — set -euo pipefail

     File: strict_demo.sh

           #!/bin/bash

set -euo pipefail

echo "=== Strict Mode Demo ==="

echo

echo "Demonstrating set -u (undefined variables)"

if [[ -z "${MY_VAR:-}" ]]; then

  echo "MY_VAR is undefined → handled safely using \${VAR:-}"

else

  echo "MY_VAR value: $MY_VAR"

fi

echo

echo "Demonstrating set -e (command failure)"

if ls /this/path/does/not/exist >/dev/null 2>&1; then

  echo "Command succeeded"

else

  echo "Command failed → detected safely using if"

fi

echo

echo "Demonstrating set -o pipefail (pipeline failure)"

if echo "hello" | grep "hello" | grep "world" >/dev/null; then

  echo "Pipeline succeeded"

else

  echo "Pipeline failed → one command in the pipeline failed"

fi

echo

echo "Script completed without crashing"    

Output:

Explanation :

·        set -e → Exit immediately if any command fails

·        set -u → Exit if an undefined variable is used

·        set -o pipefail → Fail the script if any command in a pipeline fails

## Task 5: Build Script — System Info Reporter

     File: system_info.sh

           #!/bin/bash

set -euo pipefail

print_header() {

  echo "=============================="

  echo "$1"

  echo "=============================="

}

system_info() {

  print_header "System Information"

  hostname

  uname -a

}

uptime_info() {

  print_header "Uptime"

  uptime

}

disk_usage() {

  print_header "Top 5 Disk Usage"

  du -h / 2>/dev/null | sort -hr | head -5

}

memory_usage() {

  print_header "Memory Usage"

  free -h

}

cpu_processes() {

  print_header "Top 5 CPU Consuming Processes"

  ps -eo pid,comm,%cpu --sort=-%cpu | head -6

}

main() {

  system_info

  uptime_info

  disk_usage

  memory_usage

  cpu_processes

}
